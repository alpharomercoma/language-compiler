// Function to sum numbers from 1 to n
function sum(n) {
  let total = 0;
  for (let i = 1; i <= n; i = i + 1) {
    total = total + i;
  }
  return total;
}

// Count from 1 to 10
for (let i = 1; i <= 10; i = i + 1) {
  print i;
}

// Calculate and print sum
let result = sum(10);
print "Sum of numbers from 1 to 10:";
print result;
